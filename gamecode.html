public static void game(){

    card card1 = new card(100,10,100,10);
    card card2 = new card(100,10,10,10);
    card card3 = new card(1,1,1,100);
    card card4 = new card(100,1,1,100);
    card card5 = new card(1,100,10,100);
    card card6 = new card(1,100,100,1);
    card card7 = new card(1,100,1,100);
    card card8 = new card(100,10,100,100);
    card card9 = new card(10,1,10,10);
    card card10 = new card(1,1,1,1);
    card card11 = new card(10,1,10,100);
    card card12 = new card(100,10,10,100);

    card [] cardArray = {card1,card2,card3,card4,card5,card6,card7,card8,card9,card10,card11,card12};

    final int ALL_DIFF_SUM = 111;
    final Set ALL_SAME_SUM = new HashSet();
    ALL_SAME_SUM.add(3);
    ALL_SAME_SUM.add(30);
    ALL_SAME_SUM.add(300);

    int count = 0;

    int size = cardArray.length;
    for(int i = 0 ; i < size ; i++){
        for(int j = i + 1 ; j < size; j++){
            for(int k = j + 1 ; k < size; k++){

                int shapeSum = cardArray[i].shape + cardArray[j].shape + cardArray[k].shape;
                int colorSum = cardArray[i].color + cardArray[j].color + cardArray[k].color;
                int countSum = cardArray[i].count + cardArray[j].count + cardArray[k].count;
                int fillSum = cardArray[i].fill + cardArray[j].fill + cardArray[k].fill;


                int diffcount = 0;
                int samecount = 0;
                if(ALL_DIFF_SUM == shapeSum){
                    diffcount ++;
                }
                else if(ALL_SAME_SUM.contains(shapeSum)){
                    samecount ++;
                }

                if(ALL_DIFF_SUM == colorSum){
                    diffcount ++;
                }
                else if(ALL_SAME_SUM.contains(colorSum)){
                    samecount ++;
                }

                if(ALL_DIFF_SUM == countSum){
                    diffcount ++;
                }
                else if(ALL_SAME_SUM.contains(countSum)){
                    samecount ++;
                }

                if(ALL_DIFF_SUM == fillSum){
                    diffcount ++;
                }
                else if(ALL_SAME_SUM.contains(fillSum)){
                    samecount ++;
                }

                if(diffcount + samecount == 4){
                    // 全部不同
                    if(ALL_DIFF_SUM == shapeSum
                            && ALL_DIFF_SUM == colorSum
                            && ALL_DIFF_SUM == countSum
                            && ALL_DIFF_SUM == fillSum ){
                        System.out.print("bingo:ALL DIFF"
                                + "\ncard1:" + cardArray[i].toString()
                                + "\ncard2:" + cardArray[k].toString()
                                + "\ncard3:" + cardArray[j].toString());
                        return;
                    }

                    // 全部相同
                    if(ALL_SAME_SUM.contains(shapeSum)
                            && ALL_SAME_SUM.contains(colorSum)
                            && ALL_SAME_SUM.contains(countSum)
                            && ALL_SAME_SUM.contains(fillSum) ){
                        System.out.print("bingo:ALL SAME "
                                + "\ncard1:" + cardArray[i].toString()
                                + "\ncard2:" + cardArray[k].toString()
                                + "\ncard3:" + cardArray[j].toString());
                        return;
                    }

                    // 只有一种相同
                    if(diffcount == 3){
                        System.out.print("bingo:ONLY ONE SAME"
                                + "\ncard1:" + cardArray[i].toString()
                                + "\ncard2:" + cardArray[k].toString()
                                + "\ncard3:" + cardArray[j].toString());
                        return;
                    }

                    // 只有一种不同
                    if(samecount == 3){
                        System.out.print("bingo:ONLY ONE DIFF"
                                + "\ncard1:" + cardArray[i].toString()
                                + "\ncard2:" + cardArray[k].toString()
                                + "\ncard3:" + cardArray[j].toString());
                        return;
                    }
                }
                count ++;
            }
        }
    }

    System.out.print("count:" + count);
}
